def transactions_per_sessions(value, column, df):
    total = df[df[column] == value].shape[0]
    valid_transactions = df[(df[column] == value) & (df.transactions > 0)].shape[0]
    p = valid_transactions/total
    display(f'== {value}, total sessions = {total}')
    display(f'   sessions with total transactions > 0 = {valid_transactions}')
    display(f'   percentage = {p}')

def train_evaluate(model, X_train = X_train, y_train = y_train, X_test = X_test, y_test = y_test):
    result = dict()
    classifier = model.fit(X_train, y_train)
    result['model'] = classifier
    result['pred_train'] = classifier.predict(X_train)
    result['pred_test'] = classifier.predict(X_test)
    result['f1_score_train'] = f1_score(y_train, result['pred_train'])
    result['f1_score_test'] = f1_score(y_test, result['pred_test'])
    
    cm = confusion_matrix(y_test, result['pred_test'])
    #Recall = True Positive /(True Positive + False Negative)
    result['recall'] = cm[1, 1]/(cm[1, 1] + cm[1, 0])
    #Precision = True Positive / (True Positive + False Positive)
    result['precision'] = cm[1, 1]/(cm[1, 1] + cm[0, 1])

    return result
